Create the EMPLOYEES Table:

command 1 ==>	CREATE TABLE EMPLOYEES (
		EMP_ID NUMBER PRIMARY KEY,
    		EMP_NAME VARCHAR2(100),
    		SALARY NUMBER
		);

Create the EMPLOYEE_AUDIT Table:



command 3 ==>	CREATE TABLE EMPLOYEE_AUDIT (
    		AUDIT_ID NUMBER PRIMARY KEY,
   		EMP_ID NUMBER,
    		OLD_SALARY NUMBER,
    		NEW_SALARY NUMBER,
    		CHANGE_DATE TIMESTAMP DEFAULT CURRENT_TIMESTAMP
		);

Create the Sequence:

command 4 ==>	CREATE SEQUENCE EMPLOYEE_AUDIT_SEQ
		START WITH 1
		INCREMENT BY 1;



Create the Trigger:


command 5 ==>	CREATE OR REPLACE TRIGGER EMPLOYEE_AUDIT_TRIGGER
		BEFORE INSERT ON EMPLOYEE_AUDIT
		FOR EACH ROW
		BEGIN
    		SELECT EMPLOYEE_AUDIT_SEQ.NEXTVAL INTO :NEW.AUDIT_ID FROM dual;
		END;
